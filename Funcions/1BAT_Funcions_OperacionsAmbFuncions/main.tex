\section*{1. Operacions aritmètiques i dominis}
Donades les funcions $f(x) = \frac{x^2-16}{x+1}$ i $g(x) = \frac{x+4}{x-4}$, calcula les següents operacions simplificant el resultat al màxim i indicant el \textbf{domini} de la funció resultant:

\begin{enumerate}[label=\alph*)]
    \item $(f+g)(x)$ i $(f-g)(x)$
    \item $(f \cdot g)(x)$ 
    \item $\left(\frac{f}{g}\right)(x)$
\end{enumerate}

\section*{2. Composició de funcions}
\textbf{Exemple:} Si $u(x) = x+2$ i $v(x) = x^2$, llavors $(v \circ u)(x) = v(u(x)) = (x+2)^2 = x^2 + 4x + 4$.

\vspace{0.3cm}
Donades les funcions $p(x) = 2x - 5$, $q(x) = x^2 - 3$ i $r(x) = \sqrt{x}$, troba:
\begin{enumerate}[label=\alph*)]
    \item $(q \circ p)(x)$ 
    \item $(p \circ q)(x)$ 
    \item $(r \circ q)(x)$ 
    \item $(p \circ p)(x)$
\end{enumerate}

\section*{3. Funció inversa i recorregut}
\textbf{Exemple:} Per trobar la inversa de $f(x) = 3x - 2$:
1. Escrivim $y = 3x - 2$.
2. Intercanviem variables: $x = 3y - 2$.
3. Aïllem $y$: $x+2 = 3y \rightarrow y = \frac{x+2}{3}$. Per tant, $f^{-1}(x) = \frac{x+2}{3}$.

\vspace{0.3cm}
Troba la funció inversa i el recorregut ($Im$) de les funcions següents:
\begin{enumerate}[label=\alph*)]
    \item $s(x) = 4x + 1$
    \item $h(x) = \frac{2x + 3}{x - 1}$
\end{enumerate}

\section*{Solucions dels exercicis}

\textbf{Exercici 1:}
\begin{itemize}
    \item \textbf{a)} $(f+g)(x) = \frac{x^3-3x^2-11x+68}{(x+1)(x-4)}$; $Dom = \mathbb{R} \setminus \{-1, 4\}$. $(f-g)(x) = \frac{x^3-5x^2-21x+60}{(x+1)(x-4)}$; $Dom = \mathbb{R} \setminus \{-1, 4\}$.
    \item \textbf{b)} $(f \cdot g)(x) = \frac{(x+4)^2}{x+1}$; $Dom = \mathbb{R} \setminus \{-1, 4\}$ (el domini és la intersecció dels originals).
    \item \textbf{c)} $\left(\frac{f}{g}\right)(x) = \frac{(x-4)^2}{x+1}$; $Dom = \mathbb{R} \setminus \{-1, 4, -4\}$ (cal excloure quan $g(x)=0$).
\end{itemize}

\textbf{Exercici 2:}
\begin{itemize}
    \item \textbf{a)} $4x^2 - 20x + 22$ \quad \textbf{b)} $2x^2 - 11$ \quad \textbf{c)} $\sqrt{x^2-3}$ \quad \textbf{d)} $4x - 15$
\end{itemize}

\textbf{Exercici 3:}
\begin{itemize}
    \item \textbf{a)} $s^{-1}(x) = \frac{x-1}{4}$; $Im(s) = \mathbb{R}$.
    \item \textbf{b)} $h^{-1}(x) = \frac{x+3}{x-2}$; $Im(h) = \mathbb{R} \setminus \{2\}$.
\end{itemize}
