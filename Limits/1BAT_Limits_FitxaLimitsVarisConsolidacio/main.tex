\section*{SOLUCIONS}

\subsection*{Estratègies de resolució}
\begin{itemize}
    \item \textbf{Tipus $\infty/\infty$:} Comparem el grau del numerador ($n$) i del denominador ($m$). Si $n=m$, el límit és el quocient dels coeficients principals. Si $n>m$, tendeix a $\pm\infty$. Si $n<m$, tendeix a $0$.
    \item \textbf{Tipus $1^\infty$:} S'aplica la fórmula $\displaystyle e^{\,\lim\, \text{exponent}\,\cdot\,(\text{base}-1)}$.
    \item \textbf{Tipus $0/0$:} Cal factoritzar (Ruffini o identitats notables) i simplificar. Si hi ha arrels, multipliquem pel conjugat.
    \item \textbf{Tipus $\infty-\infty$:} Multipliquem pel conjugat o posem factor comú.
    \item \textbf{Tipus $k/0$:} El límit no existeix (o és $\pm\infty$); cal estudiar els laterals.
\end{itemize}

\subsection*{Solucions Exercici 6}
\begin{enumerate}[label=6.\arabic*.]
    \item $\displaystyle \lim_{x \to 0} \frac{x^2+3x+2}{x^2-x} = \left[ \frac{2}{0} \right]$
    \begin{itemize}
        \item $\displaystyle \lim_{x \to 0^-} \frac{x^2+3x+2}{x(x-1)} = \frac{+}{(-)\cdot(-)} = +\infty$
        \item $\displaystyle \lim_{x \to 0^+} \frac{x^2+3x+2}{x(x-1)} = \frac{+}{(+)\cdot(-)} = -\infty$
        \item \textbf{Resultat: No existeix el límit.}
    \end{itemize}

    \item $\displaystyle \lim_{x \to 2} \frac{x^2+5x+6}{x^2-x-2} = \left[ \frac{20}{0} \right]$
    \begin{itemize}
        \item $\displaystyle \lim_{x \to 2^-} \frac{x^2+5x+6}{(x-2)(x+1)} = \frac{+}{(-)\cdot(+)} = -\infty$
        \item $\displaystyle \lim_{x \to 2^+} \frac{x^2+5x+6}{(x-2)(x+1)} = \frac{+}{(+)\cdot(+)} = +\infty$
        \item \textbf{Resultat: No existeix el límit.}
    \end{itemize}

    \item $\displaystyle \lim_{x \to 0} \frac{x^2+6x+8}{x^2} = \left[ \frac{8}{0} \right]$
    \begin{itemize}
        \item $\displaystyle \lim_{x \to 0^-} \frac{x^2+6x+8}{x^2} = \frac{8}{0^+} = +\infty$
        \item $\displaystyle \lim_{x \to 0^+} \frac{x^2+6x+8}{x^2} = \frac{8}{0^+} = +\infty$
        \item \textbf{Resultat: } $\mathbf{+\infty}$
    \end{itemize}

    \item $\displaystyle \lim_{x \to -1} \frac{x^2-5x+6}{x^2+2x+1} = \left[ \frac{12}{0} \right]$
    \begin{itemize}
        \item Denominador: $(x+1)^2$, sempre és positiu o zero.
        \item $\displaystyle \lim_{x \to -1^-} \frac{x^2-5x+6}{(x+1)^2} = \frac{12}{0^+} = +\infty$
        \item $\displaystyle \lim_{x \to -1^+} \frac{x^2-5x+6}{(x+1)^2} = \frac{12}{0^+} = +\infty$
        \item \textbf{Resultat: } $\mathbf{+\infty}$
    \end{itemize}

    \item $\displaystyle \lim_{x \to 4} \frac{x^2-3x+2}{x^2-x-12} = \left[ \frac{6}{0} \right]$
    \begin{itemize}
        \item $\displaystyle \lim_{x \to 4^-} \frac{x^2-3x+2}{(x-4)(x+3)} = \frac{6}{(-)\cdot(+)} = -\infty$
        \item $\displaystyle \lim_{x \to 4^+} \frac{x^2-3x+2}{(x-4)(x+3)} = \frac{6}{(+)\cdot(+)} = +\infty$
        \item \textbf{Resultat: No existeix el límit.}
    \end{itemize}

    \item $\displaystyle \lim_{x \to 1} \frac{x^3-4x^2}{x^2-4x+3} = \left[ \frac{-3}{0} \right]$
    \begin{itemize}
        \item $\displaystyle \lim_{x \to 1^-} \frac{x^3-4x^2}{(x-1)(x-3)} = \frac{-3}{(-)\cdot(-)} = -\infty$
        \item $\displaystyle \lim_{x \to 1^+} \frac{x^3-4x^2}{(x-1)(x-3)} = \frac{-3}{(+)\cdot(-)} = +\infty$
        \item \textbf{Resultat: No existeix el límit.}
    \end{itemize}
\end{enumerate}

\subsection*{Solucions de l'Exercici 7}

\begin{enumerate}[label=\textbf{7.\arabic*.}]
    \item $\lim_{x \to -1} \frac{3x^2+x-2}{x^3+x^2-x-1}$
        \begin{itemize}
            \item \textbf{Explicació:} Indeterminació $\frac{0}{0}$. Factoritzem numerador i denominador.
            \item \textbf{Resolució:}
            $$ \lim_{x \to -1} \frac{(x+1)(3x-2)}{(x-1)(x+1)^2} = \lim_{x \to -1} \frac{3x-2}{(x-1)(x+1)} = \left[ \frac{-5}{0} \right] $$
            $$ \lim_{x \to -1^-} f(x) = -\infty; \quad \lim_{x \to -1^+} f(x) = +\infty $$
            \item \textbf{Resultat:} No existeix el límit.
        \end{itemize}

    \item $\lim_{x \to 3} \left[ \frac{2x}{x^2-9} - \frac{x+1}{x-3} \right]$
        \begin{itemize}
            \item \textbf{Explicació:} Indeterminació $\infty - \infty$. Fem comú denominador $(x-3)(x+3)$.
            \item \textbf{Resolució:}
            $$ \lim_{x \to 3} \frac{2x - (x+1)(x+3)}{x^2-9} = \lim_{x \to 3} \frac{-x^2-2x-3}{x^2-9} = \left[ \frac{-18}{0} \right] $$
            \item \textbf{Resultat:} No existeix (laterals $+\infty$ i $-\infty$).
        \end{itemize}

    \item $\lim_{x \to 1} \left( \frac{2x+4}{x^2-x+6} \right)^{\frac{3x}{x-1}}$
        \begin{itemize}
            \item \textbf{Explicació:} Indeterminació $1^\infty$. Transformem la base (sumant i restant 1) per buscar la forma $\left(1+\frac{1}{f(x)}\right)^{f(x)}=e$.
            \item \textbf{Resolució:}
            $$ \lim_{x \to 1} \left( 1 + \frac{2x+4}{x^2-x+6} - 1 \right)^{\frac{3x}{x-1}} = \lim_{x \to 1} \left( 1 + \frac{-x^2+3x-2}{x^2-x+6} \right)^{\frac{3x}{x-1}} $$
            $$ = \lim_{x \to 1} \left[ \left( 1 + \frac{1}{\frac{x^2-x+6}{-x^2+3x-2}} \right)^{\frac{x^2-x+6}{-x^2+3x-2}} \right]^{\frac{-x^2+3x-2}{x^2-x+6} \cdot \frac{3x}{x-1}} $$
            $$ = e^{\lim_{x \to 1} \frac{-(x-1)(x-2) \cdot 3x}{(x^2-x+6)(x-1)}} = e^{\lim_{x \to 1} \frac{-3x(x-2)}{x^2-x+6}} = e^{3/6} = e^{1/2} $$
            \item \textbf{Resultat:} $\sqrt{e}$
        \end{itemize}

    \item $\lim_{x \to 2} \left( \frac{3x-2}{x^2-2x+4} \right)^{\frac{x}{x-2}}$
        \begin{itemize}
            \item \textbf{Explicació:} Indeterminació $1^\infty$. Busquem el número $e$.
            \item \textbf{Resolució:}
            $$ \lim_{x \to 2} \left( 1 + \frac{3x-2}{x^2-2x+4} - 1 \right)^{\frac{x}{x-2}} = \lim_{x \to 2} \left( 1 + \frac{-x^2+5x-6}{x^2-2x+4} \right)^{\frac{x}{x-2}} $$
            $$ = \lim_{x \to 2} \left[ \left( 1 + \frac{1}{\frac{x^2-2x+4}{-x^2+5x-6}} \right)^{\frac{x^2-2x+4}{-x^2+5x-6}} \right]^{\frac{-x^2+5x-6}{x^2-2x+4} \cdot \frac{x}{x-2}} $$
            $$ = e^{\lim_{x \to 2} \frac{-(x-2)(x-3) \cdot x}{(x^2-2x+4)(x-2)}} = e^{\lim_{x \to 2} \frac{-x(x-3)}{x^2-2x+4}} = e^{2/4} = e^{1/2} $$
            \item \textbf{Resultat:} $\sqrt{e}$
        \end{itemize}

    \item $\lim_{x \to 3} \left( \frac{2x^2-x+1}{4x+4} \right)^{\frac{2x}{x-3}}$
        \begin{itemize}
            \item \textbf{Explicació:} Indeterminació $1^\infty$. Busquem el número $e$.
            \item \textbf{Resolució:}
            $$ \lim_{x \to 3} \left( 1 + \frac{2x^2-x+1}{4x+4} - 1 \right)^{\frac{2x}{x-3}} = \lim_{x \to 3} \left( 1 + \frac{2x^2-5x-3}{4x+4} \right)^{\frac{2x}{x-3}} $$
            $$ = \lim_{x \to 3} \left[ \left( 1 + \frac{1}{\frac{4x+4}{2x^2-5x-3}} \right)^{\frac{4x+4}{2x^2-5x-3}} \right]^{\frac{2x^2-5x-3}{4x+4} \cdot \frac{2x}{x-3}} $$
            $$ = e^{\lim_{x \to 3} \frac{(x-3)(2x+1) \cdot 2x}{4(x+1)(x-3)}} = e^{\lim_{x \to 3} \frac{2x(2x+1)}{4(x+1)}} = e^{42/16} = e^{21/8} $$
            \item \textbf{Resultat:} $e^{21/8}$
        \end{itemize}

    \item $\lim_{x \to 0} \left( \frac{x^2-3x+1}{5x+1} \right)^{\frac{3}{x}}$
        \begin{itemize}
            \item \textbf{Explicació:} Indeterminació $1^\infty$. Busquem el número $e$.
            \item \textbf{Resolució:}
            $$ \lim_{x \to 0} \left( 1 + \frac{x^2-3x+1}{5x+1} - 1 \right)^{\frac{3}{x}} = \lim_{x \to 0} \left( 1 + \frac{x^2-8x}{5x+1} \right)^{\frac{3}{x}} $$
            $$ = \lim_{x \to 0} \left[ \left( 1 + \frac{1}{\frac{5x+1}{x^2-8x}} \right)^{\frac{5x+1}{x^2-8x}} \right]^{\frac{x^2-8x}{5x+1} \cdot \frac{3}{x}} $$
            $$ = e^{\lim_{x \to 0} \frac{x(x-8) \cdot 3}{(5x+1) \cdot x}} = e^{\lim_{x \to 0} \frac{3(x-8)}{5x+1}} = e^{-24/1} = e^{-24} $$
            \item \textbf{Resultat:} $e^{-24}$
        \end{itemize}

    \item $\lim_{x \to 1} \left( \frac{x^2-2x+3}{x+1} \right)^{\frac{1}{x-1}}$
        \begin{itemize}
            \item \textbf{Explicació:} Indeterminació $1^\infty$. Busquem el número $e$.
            \item \textbf{Resolució:}
            $$ \lim_{x \to 1} \left( 1 + \frac{x^2-2x+3}{x+1} - 1 \right)^{\frac{1}{x-1}} = \lim_{x \to 1} \left( 1 + \frac{x^2-3x+2}{x+1} \right)^{\frac{1}{x-1}} $$
            $$ = \lim_{x \to 1} \left[ \left( 1 + \frac{1}{\frac{x+1}{x^2-3x+2}} \right)^{\frac{x+1}{x^2-3x+2}} \right]^{\frac{x^2-3x+2}{x+1} \cdot \frac{1}{x-1}} $$
            $$ = e^{\lim_{x \to 1} \frac{(x-1)(x-2) \cdot 1}{(x+1)(x-1)}} = e^{\lim_{x \to 1} \frac{x-2}{x+1}} = e^{-1/2} $$
            \item \textbf{Resultat:} $e^{-1/2} = \frac{1}{\sqrt{e}}$
        \end{itemize}

    \item $\lim_{x \to +\infty} \left( \frac{1+2x}{2x+5} \right)^{2x^2-1}$
        \begin{itemize}
            \item \textbf{Explicació:} Indeterminació $1^\infty$. Busquem el número $e$.
            \item \textbf{Resolució:}
            $$ \lim_{x \to +\infty} \left( 1 + \frac{1+2x}{2x+5} - 1 \right)^{2x^2-1} = \lim_{x \to +\infty} \left( 1 + \frac{-4}{2x+5} \right)^{2x^2-1} $$
            $$ = \lim_{x \to +\infty} \left[ \left( 1 + \frac{1}{\frac{2x+5}{-4}} \right)^{\frac{2x+5}{-4}} \right]^{\frac{-4}{2x+5} \cdot (2x^2-1)} $$

    \item $\lim_{x \to -\infty} \left( \frac{4x-2}{3x+5} \right)^{x^2-1}$
        \begin{itemize}
            \item \textbf{Explicació:} La base tendeix a $4/3 > 1$ i l'exponent a $+\infty$. No és indeterminació.
            \item \textbf{Resultat:} $+\infty$

    \item $\lim_{x \to +\infty} \left( \frac{2x-2}{3+2x} \right)^{x+1}$

\section*{Resolucions pas per pas}



\begin{enumerate}[label=\textbf{3.\arabic*.}, leftmargin=*, itemsep=0.8cm]

    % 3.1
    \item $\displaystyle \lim_{x \to 2} \frac{x^2+x-6}{x^2-x-2}$ \\
    Factoritzem els polinomis de segon grau buscant les seves arrels:
    \begin{align*}
        \lim_{x \to 2} \frac{(x-2)(x+3)}{(x-2)(x+1)} = \lim_{x \to 2} \frac{x+3}{x+1} = \frac{2+3}{2+1} = \mathbf{\frac{5}{3}}
    \end{align*}

    % 3.2
    \item $\displaystyle \lim_{x \to 3} \frac{x^2-6x+9}{x-3}$ \\
    El numerador és una identitat notable (quadrat d'una resta):
    \begin{align*}
        \lim_{x \to 3} \frac{(x-3)^2}{x-3} = \lim_{x \to 3} (x-3) = 3 - 3 = \mathbf{0}
    \end{align*}

    % 3.3
    \item $\displaystyle \lim_{x \to 1} \frac{x^2-1}{x^3-1}$ \\
    Factoritzem el numerador (suma per diferència) i el denominador (Ruffini):
    \begin{align*}
        \lim_{x \to 1} \frac{(x-1)(x+1)}{(x-1)(x^2+x+1)} = \lim_{x \to 1} \frac{x+1}{x^2+x+1} = \frac{1+1}{1+1+1} = \mathbf{\frac{2}{3}}
    \end{align*}

    % 3.4
    \item $\displaystyle \lim_{x \to -2} \frac{x^2-x-6}{x^3+3x^2+2x}$ \\
    Traiem factor comú de la $x$ al denominador i factoritzem la resta:
    \begin{align*}
        \lim_{x \to -2} \frac{(x+2)(x-3)}{x(x^2+3x+2)} &= \lim_{x \to -2} \frac{(x+2)(x-3)}{x(x+2)(x+1)} \\
        &= \lim_{x \to -2} \frac{x-3}{x(x+1)} = \frac{-2-3}{-2(-2+1)} = \frac{-5}{2} = \mathbf{-\frac{5}{2}}
    \end{align*}

    % 3.5
    \item $\displaystyle \lim_{x \to 1} \sqrt[3]{\frac{2x^3-3x^2+1}{3x^3-8x^2+7x-2}}$ \\
    Apliquem Ruffini dos cops a cada polinomi per trobar l'arrel $x=1$ repetida:
    \begin{align*}
        \lim_{x \to 1} \sqrt[3]{\frac{(x-1)^2(2x+1)}{(x-1)^2(3x-2)}} = \lim_{x \to 1} \sqrt[3]{\frac{2x+1}{3x-2}} = \sqrt[3]{\frac{2(1)+1}{3(1)-2}} = \sqrt[3]{\frac{3}{1}} = \mathbf{\sqrt[3]{3}}
    \end{align*}

    % 3.6
    \item $\displaystyle \lim_{x \to 1} \frac{x^2-2x+1}{x-1}$ \\
    El numerador és una identitat notable:
    \begin{align*}
        \lim_{x \to 1} \frac{(x-1)^2}{x-1} = \lim_{x \to 1} (x-1) = 1 - 1 = \mathbf{0}
    \end{align*}

    % 3.7
    \item $\displaystyle \lim_{x \to 0} \frac{x}{1-\sqrt{1-x}}$ \\
    Multipliquem numerador i denominador pel conjugat $1+\sqrt{1-x}$:
    \begin{align*}
        \lim_{x \to 0} \frac{x(1+\sqrt{1-x})}{(1-\sqrt{1-x})(1+\sqrt{1-x})} &= \lim_{x \to 0} \frac{x(1+\sqrt{1-x})}{1 - (1-x)} \\
        &= \lim_{x \to 0} \frac{x(1+\sqrt{1-x})}{x} = \lim_{x \to 0} (1+\sqrt{1-x}) = 1+\sqrt{1} = \mathbf{2}
    \end{align*}

    % 3.8
    \item $\displaystyle \lim_{x \to 3} \frac{\sqrt{x+1}-2}{x-3}$ \\
    Multipliquem pel conjugat $\sqrt{x+1}+2$:
    \begin{align*}
        \lim_{x \to 3} \frac{(\sqrt{x+1}-2)(\sqrt{x+1}+2)}{(x-3)(\sqrt{x+1}+2)} &= \lim_{x \to 3} \frac{(x+1)-4}{(x-3)(\sqrt{x+1}+2)} \\
        &= \lim_{x \to 3} \frac{x-3}{(x-3)(\sqrt{x+1}+2)} = \lim_{x \to 3} \frac{1}{\sqrt{x+1}+2} = \frac{1}{\sqrt{4}+2} = \mathbf{\frac{1}{4}}
    \end{align*}

    % 3.9
    \item $\displaystyle \lim_{x \to 0} \frac{\sqrt{x+9}-3}{\sqrt{x+16}-4}$ \\
    Doble conjugat! Multipliquem pels conjugats del numerador i del denominador:
    \begin{align*}
        &\lim_{x \to 0} \frac{(\sqrt{x+9}-3)(\sqrt{x+9}+3)(\sqrt{x+16}+4)}{(\sqrt{x+16}-4)(\sqrt{x+16}+4)(\sqrt{x+9}+3)} \\
        &= \lim_{x \to 0} \frac{(x+9-9)(\sqrt{x+16}+4)}{(x+16-16)(\sqrt{x+9}+3)} = \lim_{x \to 0} \frac{x(\sqrt{x+16}+4)}{x(\sqrt{x+9}+3)} \\
        &= \lim_{x \to 0} \frac{\sqrt{x+16}+4}{\sqrt{x+9}+3} = \frac{4+4}{3+3} = \frac{8}{6} = \mathbf{\frac{4}{3}}
    \end{align*}

    % 3.10
    \item $\displaystyle \lim_{x \to 0} \frac{\sqrt{2x+4}-2}{\sqrt{x+1}-1}$ \\
    Un altre doble conjugat:
    \begin{align*}
        &\lim_{x \to 0} \frac{(\sqrt{2x+4}-2)(\sqrt{2x+4}+2)(\sqrt{x+1}+1)}{(\sqrt{x+1}-1)(\sqrt{x+1}+1)(\sqrt{2x+4}+2)} \\
        &= \lim_{x \to 0} \frac{(2x+4-4)(\sqrt{x+1}+1)}{(x+1-1)(\sqrt{2x+4}+2)} = \lim_{x \to 0} \frac{2x(\sqrt{x+1}+1)}{x(\sqrt{2x+4}+2)} \\
        &= \lim_{x \to 0} \frac{2(\sqrt{x+1}+1)}{\sqrt{2x+4}+2} = \frac{2(1+1)}{2+2} = \frac{4}{4} = \mathbf{1}
    \end{align*}

    % 3.11
    \item $\displaystyle \lim_{x \to 0} \frac{2-\sqrt{4-x}}{x}$ \\
    Multipliquem pel conjugat $2+\sqrt{4-x}$:
    \begin{align*}
        \lim_{x \to 0} \frac{(2-\sqrt{4-x})(2+\sqrt{4-x})}{x(2+\sqrt{4-x})} &= \lim_{x \to 0} \frac{4 - (4-x)}{x(2+\sqrt{4-x})} \\
        &= \lim_{x \to 0} \frac{x}{x(2+\sqrt{4-x})} = \lim_{x \to 0} \frac{1}{2+\sqrt{4-x}} = \frac{1}{2+2} = \mathbf{\frac{1}{4}}
    \end{align*}

    % 3.12
    \item $\displaystyle \lim_{x \to 1} \frac{x-1}{\sqrt{x}-1}$ \\
    Multipliquem pel conjugat del denominador:
    \begin{align*}
        \lim_{x \to 1} \frac{(x-1)(\sqrt{x}+1)}{(\sqrt{x}-1)(\sqrt{x}+1)} &= \lim_{x \to 1} \frac{(x-1)(\sqrt{x}+1)}{x-1} \\
        &= \lim_{x \to 1} (\sqrt{x}+1) = \sqrt{1}+1 = \mathbf{2}
    \end{align*}

    % 3.13
    \item $\displaystyle \lim_{x \to 0} \frac{x}{\sqrt{1+x}-\sqrt{1-x}}$ \\
    Multipliquem pel conjugat del denominador:
    \begin{align*}
        \lim_{x \to 0} \frac{x(\sqrt{1+x}+\sqrt{1-x})}{(\sqrt{1+x}-\sqrt{1-x})(\sqrt{1+x}+\sqrt{1-x})} &= \lim_{x \to 0} \frac{x(\sqrt{1+x}+\sqrt{1-x})}{(1+x) - (1-x)} \\
        &= \lim_{x \to 0} \frac{x(\sqrt{1+x}+\sqrt{1-x})}{2x} = \lim_{x \to 0} \frac{\sqrt{1+x}+\sqrt{1-x}}{2} \\
        &= \frac{\sqrt{1}+\sqrt{1}}{2} = \frac{2}{2} = \mathbf{1}
    \end{align*}

    % 3.14
    \item $\displaystyle \lim_{x \to 5} \frac{\sqrt{x}-\sqrt{5}}{x-5}$ \\
    Multipliquem pel conjugat del numerador:
    \begin{align*}
        \lim_{x \to 5} \frac{(\sqrt{x}-\sqrt{5})(\sqrt{x}+\sqrt{5})}{(x-5)(\sqrt{x}+\sqrt{5})} &= \lim_{x \to 5} \frac{x-5}{(x-5)(\sqrt{x}+\sqrt{5})} \\
        &= \lim_{x \to 5} \frac{1}{\sqrt{x}+\sqrt{5}} = \frac{1}{2\sqrt{5}} = \mathbf{\frac{\sqrt{5}}{10}} \text{ (racionalitzat)}
    \end{align*}

    % 3.15
    \item $\displaystyle \lim_{x \to 1} \frac{x^3-1}{x^2-1}$ \\
    Factoritzem numerador (Ruffini o diferència de cubs) i denominador:
    \begin{align*}
        \lim_{x \to 1} \frac{(x-1)(x^2+x+1)}{(x-1)(x+1)} = \lim_{x \to 1} \frac{x^2+x+1}{x+1} = \frac{1+1+1}{1+1} = \mathbf{\frac{3}{2}}
    \end{align*}

    % 3.16
    \item $\displaystyle \lim_{x \to 2} \frac{-2x^2+5x-2}{3x^2-2x-8}$ \\
    Busquem les arrels amb l'equació de segon grau. Arrels del numerador: $2$ i $1/2$. Arrels del denominador: $2$ i $-4/3$. Factoritzem:
    \begin{align*}
        \lim_{x \to 2} \frac{-2(x-2)(x-1/2)}{3(x-2)(x+4/3)} &= \lim_{x \to 2} \frac{-(2x-1)(x-2)}{(3x+4)(x-2)} \\
        &= \lim_{x \to 2} \frac{-2x+1}{3x+4} = \frac{-4+1}{6+4} = \mathbf{-\frac{3}{10}}
    \end{align*}

    % 3.17
    \item $\displaystyle \lim_{x \to 2} \frac{x^2+x-6}{x^2-x-2}$ \\
    Aquest límit és exactament el mateix que l'apartat 3.1.
    \begin{align*}
        \lim_{x \to 2} \frac{(x-2)(x+3)}{(x-2)(x+1)} = \lim_{x \to 2} \frac{x+3}{x+1} = \frac{2+3}{2+1} = \mathbf{\frac{5}{3}}
    \end{align*}

\end{enumerate}
