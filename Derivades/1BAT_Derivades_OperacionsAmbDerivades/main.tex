\section*{Exercicis de Derivades}

\begin{enumerate}[leftmargin=*, label=\textbf{\arabic*.}]

    % EXERCICI 1
    \item \textbf{Derivades immediates.}
    
    \textit{Recorda:} Cada funció elemental té una regla de derivació que cal saber-se. Per exemple, la derivada de la funció potencial és $f(x) = x^n \rightarrow f'(x) = n \cdot x^{n-1}$.
    
    \textbf{Exemple:} Si $f(x) = x^3 \rightarrow f'(x) = 3x^2$
    \vspace{0.2cm}
    
    Calcula la derivada de les següents funcions:
    \begin{multicols}{2}
    \begin{enumerate}[label=\alph*)]
        \item $f(x) = x^5$
        \item $f(x) = e^x$
        \item $f(x) = \ln(x)$
        \item $f(x) = \cos(x)$
        \item $f(x) = 3^x$
        \item $f(x) = \sin(x)$
        \item $f(x) = \log_5(x)$
        \item $f(x) = 7$
    \end{enumerate}
    \end{multicols}
    \vspace{0.4cm}

    % EXERCICI 2
    \item \textbf{Producte d'un nombre per una funció.}
    
    \textit{Fórmula:} $[k \cdot f(x)]' = k \cdot f'(x)$
    
    \textit{Recorda:} La derivada del producte d'un nombre per una funció és el producte del nombre per la derivada de la funció.
    
    \textbf{Exemple:} Si $f(x) = 2 \cdot x^3 \rightarrow f'(x) = 2 \cdot (3x^2) = 6x^2$
    \vspace{0.2cm}
    
    Deriva les funcions següents:
    \begin{multicols}{2}
    \begin{enumerate}[label=\alph*)]
        \item $f(x) = 4x^3$
        \item $f(x) = -2\sin(x)$
        \item $f(x) = 5\ln(x)$
        \item $f(x) = \frac{1}{2} e^x$
        \item $f(x) = 6\cos(x)$
        \item $f(x) = -3 \cdot 2^x$
    \end{enumerate}
    \end{multicols}
    \vspace{0.4cm}
    \newpage
    % EXERCICI 3
    \item \textbf{Derivada del producte de funcions.}
    
    \textit{Fórmula:} $[f(x) \cdot g(x)]' = f'(x) \cdot g(x) + f(x) \cdot g'(x)$
    
    \textbf{Exemple:} Si $f(x) = x - 1$ i $g(x) = 3^x$. Llavors, la funció $h(x) = f(x) \cdot g(x)$ té per derivada: $h'(x) = 1 \cdot 3^x + (x - 1) \cdot (3^x \cdot \ln(3))$
    \vspace{0.2cm}
    
    Troba la funció derivada:
    \begin{multicols}{2}
    \begin{enumerate}[label=\alph*)]
        \item $f(x) = x^2 \sin(x)$
        \item $f(x) = e^x \cos(x)$
        \item $f(x) = x^4 \ln(x)$
        \item $f(x) = (x^2 + 1)(x - 3)$
        \item $f(x) = 3^x \sin(x)$
        \item $f(x) = x^3 e^x$
    \end{enumerate}
    \end{multicols}
    \vspace{0.4cm}

    % EXERCICI 4
    \item \textbf{Derivada del quocient de funcions.}
    
    \textit{Fórmula:} $\displaystyle \left[\frac{f(x)}{g(x)}\right]' = \frac{f'(x) \cdot g(x) - f(x) \cdot g'(x)}{(g(x))^2}$
    
    \textbf{Exemple:} Si $h(x) = \frac{x-1}{x^2-2} \rightarrow h'(x) = \frac{1 \cdot (x^2-2) - (x-1) \cdot (2x)}{(x^2-2)^2}$
    \vspace{0.2cm}
    
    Calcula les derivades:
    \begin{multicols}{2}
    \begin{enumerate}[label=\alph*)]
        \item $f(x) = \frac{x^2 - 1}{x}$
        \item $f(x) = \frac{\sin(x)}{x^2}$
        \item $f(x) = \frac{e^x}{x+2}$
        \item $f(x) = \frac{\ln(x)}{x}$
        \item $f(x) = \frac{2x + 3}{x - 1}$
        \item $f(x) = \frac{\cos(x)}{e^x}$
    \end{enumerate}
    \end{multicols}
    \vspace{0.4cm}

    % EXERCICI 5
    \item \textbf{Operacions combinades.}
    
    \vspace{0.2cm}
    
    Troba la derivada de les funcions següents:
    \begin{multicols}{2}
    \begin{enumerate}[label=\alph*)]
        \item $f(x) = 3x^4 - 5x^2 + 2$
        \item $f(x) = 2x^3 \ln(x)$
        \item $f(x) = \frac{3x}{x^2 + 1}$
        \item $f(x) = -4\sin(x) + 2\cos(x)$
        \item $f(x) = (2x^2 - x)e^x$
        \item $f(x) = 5^x - \log_2(x)$
        \item $f(x) = \frac{1}{x^3}$
        \item $f(x) = 3x \cos(x)$
        \item $f(x) = \frac{e^x - 1}{e^x + 1}$
        \item $f(x) = x^2 \sin(x) + e^x$
    \end{enumerate}
    \end{multicols}

\end{enumerate}
