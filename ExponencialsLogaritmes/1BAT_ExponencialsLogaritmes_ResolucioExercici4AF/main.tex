\section*{Resolució pas a pas}

\begin{enumerate}[label=\textbf{\alph*)}, leftmargin=*, itemsep=1cm]

    % --- APARTAT A ---
    \item \textbf{Equació:} $x^{-4} = 256$ \\
    Convertim l'exponent negatiu en una fracció, passem $x^4$ multiplicant a l'altra banda i aïllem:
    \begin{align*}
        \frac{1}{x^4} &= 256 \\
        1 &= 256 \cdot x^4 \\
        x^4 &= \frac{1}{256} \\
        x &= \pm \sqrt[4]{\frac{1}{256}} = \pm \frac{1}{4}
    \end{align*}

    % --- APARTAT B ---
    \item \textbf{Equació:} $\sqrt{3\sqrt{3\sqrt{3\sqrt{3}}}} = \left(\frac{1}{3}\right)^{1-x}$ \\
    Anem extraient factors de dins cap a fora, convertint a exponent fraccionari i sumant els exponents amb la mateixa base. A la dreta, transformem el $\frac{1}{3}$ en $3^{-1}$:
    \begin{align*}
        \sqrt{3\sqrt{3\sqrt{3 \cdot 3^{1/2}}}} &= (3^{-1})^{1-x} \\
        \sqrt{3\sqrt{3\sqrt{3^{3/2}}}} &= 3^{-1+x} \\
        \sqrt{3\sqrt{3 \cdot 3^{3/4}}} &= 3^{x-1} \\
        \sqrt{3\sqrt{3^{7/4}}} &= 3^{x-1} \\
        \sqrt{3 \cdot 3^{7/8}} &= 3^{x-1} \\
        \sqrt{3^{15/8}} &= 3^{x-1} \\
        3^{15/16} &= 3^{x-1}
    \end{align*}
    Un cop tenim la mateixa base, igualem exponents:
    \begin{align*}
        \frac{15}{16} &= x - 1 \\
        x &= \frac{15}{16} + 1 = \mathbf{\frac{31}{16}}
    \end{align*}

    % --- APARTAT C ---
    \item \textbf{Equació:} $3^x \cdot 5^{x-1} = 10125$ \\
    Separem l'exponent de base 5 i el passem multiplicant a l'altra banda. Després, ajuntem les bases amb mateix exponent i descomponem el número resultant:
    \begin{align*}
        3^x \cdot \frac{5^x}{5} &= 10125 \\
        3^x \cdot 5^x &= 10125 \cdot 5 \\
        (3 \cdot 5)^x &= 50625 \\
        15^x &= 50625
    \end{align*}
    Descomponem $50625$: $50625 = 15^4$.
    \begin{align*}
        15^x &= 15^4 \implies \mathbf{x = 4}
    \end{align*}

    % --- APARTAT D ---
    \item \textbf{Equació:} $\sqrt{\sqrt{7} + 6\sqrt{7}} = 49^{\frac{x}{2}}$ \\
    Sumem les arrels semblants del terme de l'esquerra, ho convertim a potències d'exponent fraccionari, sumem exponents i igualem bases:
    \begin{align*}
        \sqrt{7\sqrt{7}} &= (7^2)^{\frac{x}{2}} \\
        \sqrt{7 \cdot 7^{1/2}} &= 7^x \\
        \sqrt{7^{3/2}} &= 7^x \\
        (7^{3/2})^{1/2} &= 7^x \\
        7^{3/4} &= 7^x \implies \mathbf{x = \frac{3}{4}}
    \end{align*}

    % --- APARTAT E ---
    \item \textbf{Equació:} $27x^3 = \frac{1}{125}$ \\
    Passem el $27$ dividint a l'altra banda i fem l'arrel cúbica:
    \begin{align*}
        x^3 &= \frac{1}{27 \cdot 125} \\
        x &= \sqrt[3]{\frac{1}{27 \cdot 125}} = \frac{1}{\sqrt[3]{27} \cdot \sqrt[3]{125}} \\
        x &= \frac{1}{3 \cdot 5} = \mathbf{\frac{1}{15}}
    \end{align*}

    % --- APARTAT F ---
    \item \textbf{Equació:} $5^{4x} - 3 \cdot 5^{2x} - 10 = 0$ \\
    Fem un canvi de variable per simplificar l'equació a segon grau. Sigui $t = 5^{2x}$, per tant $t^2 = (5^{2x})^2 = 5^{4x}$:
    \begin{align*}
        t^2 - 3t - 10 &= 0
    \end{align*}
    Resolem l'equació de segon grau i obtenim $t = 5$ i $t = -2$. Ara desfem el canvi de variable:
    \begin{itemize}
        \item Si $t = 5 \implies 5^{2x} = 5^1 \implies 2x = 1 \implies \mathbf{x = \frac{1}{2}}$
        \item Si $t = -2 \implies 5^{2x} = -2 \implies$ \textbf{No té solució real} (una exponencial no pot ser negativa).
    \end{itemize}

    % --- APARTAT G ---
    \item \textbf{Equació:} $5^{x-1} = 2 + \frac{3}{5^{2-x}}$ \\
    Aquesta equació equival a $5^{x-1} = 2 + 3 \cdot 5^{x-2}$. Separem els exponents i fem un denominador comú:
    \begin{align*}
        \frac{5^x}{5} &= 2 + \frac{3 \cdot 5^x}{25} \\
        \frac{5 \cdot 5^x}{25} &= \frac{50}{25} + \frac{3 \cdot 5^x}{25} \\
        5 \cdot 5^x &= 50 + 3 \cdot 5^x
    \end{align*}
    Passem tots els termes amb $5^x$ a la mateixa banda i extraiem factor comú:
    \begin{align*}
        5 \cdot 5^x - 3 \cdot 5^x &= 50 \\
        5^x (5 - 3) &= 50 \\
        5^x \cdot 2 &= 50 \\
        5^x &= 25 \\
        5^x &= 5^2 \implies \mathbf{x = 2}
    \end{align*}

    % --- APARTAT H ---
    \item \textbf{Equació:} $(a^{x-3})^x = \left(\frac{1}{a}\right)^{-2x}$ \\
    Multipliquem els exponents en la potència de potència, canviem $\frac{1}{a}$ per $a^{-1}$ i igualem:
    \begin{align*}
        a^{x(x-3)} &= (a^{-1})^{-2x} \\
        a^{x^2 - 3x} &= a^{2x}
    \end{align*}
    Com que les bases són iguals, igualem els exponents i resolem l'equació de segon grau:
    \begin{align*}
        x^2 - 3x &= 2x \\
        x^2 - 5x &= 0 \\
        x(x - 5) &= 0 \implies \mathbf{x = 0}, \mathbf{x = 5}
    \end{align*}

    % --- APARTAT I ---
    \item \textbf{Equació:} $(2x)^{-\frac{2}{5}} = \sqrt[5]{e^4}$ \\
    Primer canviem l'exponent a positiu passant la potència al denominador, separem el 2 de la $x$, agrupem la $x$ a un costat i els nombres a l'altre, elevem a 5 i apliquem arrel quadrada:
    \begin{align*}
        \frac{1}{(2x)^{2/5}} &= e^{4/5} \\
        \frac{1}{2^{2/5} \cdot x^{2/5}} &= e^{4/5} \\
        \frac{1}{2^{2/5}} &= e^{4/5} \cdot x^{2/5} \\
        \frac{1}{2^{2/5} \cdot e^{4/5}} &= x^{2/5}
    \end{align*}
    Elevem tota l'equació a la $5$ per eliminar els denominadors dels exponents:
    \begin{align*}
        \left( \frac{1}{2^{2/5} \cdot e^{4/5}} \right)^5 &= \left( x^{2/5} \right)^5 \\
        \frac{1}{2^2 \cdot e^4} &= x^2 \\
        x^2 &= \frac{1}{4e^4} \\
        x &= \pm \sqrt{\frac{1}{4e^4}} = \mathbf{\pm \frac{1}{2e^2}}
    \end{align*}

    % --- APARTAT J ---
    \item \textbf{Equació:} $7^x + 7^{x+1} + 7^{x+2} = 2793$ \\
    Descomponem les potències per fer aparèixer el $7^x$ a tots els termes i en traiem factor comú:
    \begin{align*}
        7^x + 7^x \cdot 7^1 + 7^x \cdot 7^2 &= 2793 \\
        7^x (1 + 7 + 49) &= 2793 \\
        7^x \cdot 57 &= 2793 \\
        7^x &= \frac{2793}{57} \\
        7^x &= 49 \\
        7^x &= 7^2 \implies \mathbf{x = 2}
    \end{align*}

\end{enumerate}
