% ---- Caixa general de propietats ----
\begin{metodebox}
\textbf{Propietats clau} (per a $a>0$, $a\neq 1$, $x,z > 0$):
\[
\log_a x + \log_a z = \log_a(xz) \qquad
\log_a x - \log_a z = \log_a\!\left(\tfrac{x}{z}\right) \qquad
p\cdot\log_a x = \log_a x^p \qquad
\log_a b = \frac{\log b}{\log a}
\]
\begin{verificaciobox}
\textbf{Verificació obligatòria:} Substitueix cada solució a l'equació original i comprova que:
(1) tots els arguments dels logaritmes són $>0$;\quad
(2) si la base conté la incògnita, cal que $a \in (0,1)\cup(1,+\infty)$.
\end{verificaciobox}
\end{metodebox}

\vspace{0.3cm}

\noindent{\large\textbf{\color{taronja}Pas previ \quad La definició de logaritme}}
\vspace{0.12cm}\hrule height 1pt\vspace{0.25cm}

\begin{metodebox}
La definició de logaritme: $\log_a b = c \;\Longleftrightarrow\; a^c = b$. Ens permet resoldre directament equacions del tipus:
\[
\log_a(\text{expr}) = n \;\longrightarrow\; \text{expr} = a^n
\]
\end{metodebox}

\vspace{0.2cm}

\begin{tcolorbox}[exemplebox]
Resol $\;\log_3(2x - 1) = 3$
\tcblower
\[
2x - 1 = 3^3 = 27 \;\longrightarrow\; 2x = 28 \;\longrightarrow\; \boldsymbol{x = 14}
\]
\textbf{Verificació:} $\log_3(27) = 3$ \checkmark \quad (argument $27 > 0$ \checkmark)
\end{tcolorbox}

\vspace{0.3cm}
\noindent\textbf{Practica:}

\begin{exercicis}
    \item $\log_2(x + 3) = 4$
    \item $\log_5(3x - 2) = 2$
    \item $\log(x^2 - 19) = 1$
\end{exercicis}

% =====================================================================
\noindent{\large\textbf{\color{taronja}1. Resta de logaritmes igual a un nombre}}
% =====================================================================

\begin{metodebox}
\textbf{Mètode:}
\begin{enumerate}[noitemsep, label=\arabic*.]
    \item Convertir el nombre a logaritme: $n = \log_a a^n$.
    \item Aplicar: $\log_a x - \log_a z = \log_a\!\left(\frac{x}{z}\right)$.
    \item Igualar arguments i resoldre. \quad
    \item Verificar el domini.
\end{enumerate}
\end{metodebox}

\vspace{0.2cm}

\begin{tcolorbox}[exemplebox]
Resol $\;\log(x) - \log(x-9) = 1$
\tcblower
$\log\!\left(\frac{x}{x-9}\right) = \log 10 \;\longrightarrow\;
\frac{x}{x-9} = 10 \;\longrightarrow\; x = 10x-90 \;\longrightarrow\; \boldsymbol{x = 10}$

\textbf{Verificació:} $\log 10 - \log 1 = 1$ \checkmark \quad (arguments $10>0$, $1>0$ \checkmark)
\end{tcolorbox}

\vspace{0.3cm}
\noindent\textbf{Practica:}

\begin{exercicis}
    \item $\log(5x) - \log(x-1) = 1$
    \item $\log_3(x+6) - \log_3(x) = 1$
    \item $\log_4(x) - \log_4(x-3) = 1$
    \item $2\log(x) - \log(x-6) = 2 \quad$ \textit{(Pista: aplica primer $p\cdot\log_a x = \log_a x^p$)}
    \item $\log(x) - \log(x+9) = -1$
\end{exercicis}

\newpage

% =====================================================================


\noindent{\large\textbf{\color{taronja}2. Suma de logaritmes igual a un nombre}}

% =====================================================================

\begin{metodebox}
\textbf{Mètode:}
\begin{enumerate}[noitemsep, label=\arabic*.]
    \item Convertir el nombre a logaritme: $n = \log_a a^n$.
    \item Aplicar: $\log_a x + \log_a z = \log_a(x \cdot z)$.
    \item Igualar arguments i resoldre (sovint equació de 2n grau).
    \item Verificar i descartar solucions fora del domini.
\end{enumerate}
\end{metodebox}

\vspace{0.2cm}

\begin{tcolorbox}[exemplebox]
Resol $\;\log(x) + \log(x-3) = 1$
\tcblower
$\log(x(x-3)) = \log 10 \;\longrightarrow\; x^2-3x-10=0 \;\longrightarrow\; (x-5)(x+2)=0$
\begin{itemize}[noitemsep]
    \item $x_1=5$: $\log 5 + \log 2 = 1$ \checkmark
    \item $x_2=-2$: $\log(-2)$ no existeix $\Rightarrow$ \textbf{descartada}
\end{itemize}
Solució: $\boldsymbol{x=5}$
\end{tcolorbox}

\vspace{0.3cm}
\noindent\textbf{Practica:}

\begin{exercicis}
    \item $\log_3(x) + \log_3(x-8) = 2$
    \item $\log_2(x) + \log_2(x+2) = 3$
    \item $\log(x) + \log(x+21) = 2$
    \item $\log_2(x+1) + \log_2(x-1) = 3$
\end{exercicis}

\newpage
% =====================================================================

\noindent{\large\textbf{\color{taronja}3. Quocient de logaritmes igual a un nombre}}
% =====================================================================

\begin{metodebox}
\textbf{Mètode:}
\begin{enumerate}[noitemsep, label=\arabic*.]
    \item Multiplicar els dos costats pel logaritme del denominador.
    \item Aplicar $p \cdot \log_a x = \log_a x^p$ i igualar arguments.
    \item Verificar: a més del domini, el denominador no pot ser $\log_a(1) = 0$.
\end{enumerate}
\end{metodebox}

\vspace{0.2cm}

\begin{tcolorbox}[exemplebox]
Resol $\;\dfrac{\log(x+6)}{\log(x)} = 2$
\tcblower
$\log(x+6) = 2\log(x) = \log(x^2) \;\longrightarrow\; x+6=x^2 \;\longrightarrow\; x^2-x-6=0$

$(x-3)(x+2)=0$: \; $x_1=3$ \checkmark, \; $x_2=-2$ \textbf{desc.} \quad Solució: $\boldsymbol{x=3}$
\end{tcolorbox}

\vspace{0.3cm}
\noindent\textbf{Practica:}

\begin{exercicis}
    \item $\dfrac{\log(3x-2)}{\log(x)} = 2$
    \item $\dfrac{\log(x+5)}{\log(x-1)} = 2$
    \item $\dfrac{\log(x+9)}{\log(x+3)} = 2$
    \item $\dfrac{\log(2+x^2)}{\log(3+x)} = 2$
\end{exercicis}

\newpage

% =====================================================================
\noindent{\large\textbf{\color{taronja}4. Equació exponencial --- aplicar logaritme als dos costats}}
\vspace{0.12cm}\hrule height 1pt\vspace{0.3cm}

% =====================================================================

\begin{metodebox}
\textbf{Mètode:} Quan no podem igualar bases, apliquem $\log$ als dos costats i fem servir $\log a^n = n\cdot\log a$:
\[
a^{f(x)} = b \;\longrightarrow\; f(x)\cdot\log a = \log b \;\longrightarrow\; f(x) = \frac{\log b}{\log a}
\]
\textbf{Atenció:} El resultat conté logaritmes. Cal expressar-los sempre en base 10 o $e$.
\end{metodebox}

\vspace{0.2cm}

\begin{tcolorbox}[exemplebox]
Resol $\;2^x = 6$
\tcblower
$\log 2^x = \log 6 \;\longrightarrow\; x\cdot\log 2 = \log 6 \;\longrightarrow\; \boldsymbol{x = \dfrac{\log 6}{\log 2} \approx 2{,}58}$

\textbf{Verificació:} $2^{2{,}58}\approx 6$ \checkmark
\end{tcolorbox}

\vspace{0.3cm}
\noindent\textbf{Practica:}

\begin{exercicis}
    \item $3^x = 15$
    \item $2^{3x} = 20$
    \item $5^{x-1} = 12$
    \item $4^x = 7^{x-2} \quad$ \textit{(Pista: aplica $\log$ als dos costats i desenvolupa)}
\end{exercicis}

% =====================================================================
\noindent{\large\textbf{\color{taronja}5. Altres casos: base incògnita i incògnita fora del logaritme}}
\vspace{0.12cm}\hrule height 1pt\vspace{0.3cm}
% =====================================================================

\begin{metodebox}
\textbf{Cas A --- La base conté la incògnita:} Definició inversa: $\log_{f(x)} k = n \;\longrightarrow\; (f(x))^n = k$

\textbf{Important:} Verifica que la base resultant pertanyi a $(0,1)\cup(1,+\infty)$.

\medskip
\textbf{Cas B --- La incògnita és fora de tot logaritme:} Aïlla la $x$. El logaritme queda com a terme numèric; expressa'l en base 10 amb el canvi de base.
\end{metodebox}

\vspace{0.1cm}

\begin{tcolorbox}[exemplebox, title={Exemple resolt --- Cas A: base incògnita}]
Resol $\;\log_{x+1} 9 = 2$
\tcblower
$(x+1)^2 = 9 \;\longrightarrow\; x+1=\pm 3 \;\longrightarrow\; x_1=2,\; x_2=-4$

\textbf{Verificació:} $x_1=2$: base $=3$ \checkmark;\quad $x_2=-4$: base $=-3\notin(0,1)\cup(1,+\infty)$ \textbf{desc.}

Solució: $\boldsymbol{x=2}$
\end{tcolorbox}

\vspace{0.15cm}

\begin{tcolorbox}[exemplebox, title={Exemple resolt --- Cas B: incògnita fora del logaritme}]
Resol $\;\log_2 5 = x - 1$
\tcblower
$x = 1 + \log_2 5 = 1 + \dfrac{\log 5}{\log 2} \approx 1 + 2{,}32 = \boldsymbol{3{,}32}$

\textbf{Verificació:} $\log_2 5 \approx 2{,}32 = 3{,}32 - 1$ \checkmark
\end{tcolorbox}

\vspace{0.2cm}
\noindent\textbf{Practica:}

\noindent\textit{Cas A:}
\begin{exercicis}
    \item $\log_{x-1} 16 = 4$
    \item $\log_{2x-1} 49 = 2$
\end{exercicis}

\noindent\textit{Cas B:}
\begin{exercicis}[start=3]
    \item $\log_3 10 = 2x + 1$
    \item $\log_2 12 = x + 3$
\end{exercicis}
