\secciotipus{Reconeixement directe de potĆØncies}
% ========================================================

\begin{metodebox}
\textbf{MĆØtode:} El nombre de l'altre costat de la igualtat ja Ć©s (o Ć©s fĆ cilment expressable com) una potĆØncia de la base que tenim. Reescrivim-lo, igualem els exponents i resolem l'equaciĆ³ lineal resultant.
\end{metodebox}

\vspace{0.1cm}

\begin{tcolorbox}[exemplebox]
Resol $\;2^x = 32$
\tcblower
Observem que $32 = 2^5$, per tant:
\[
2^x = 2^5 \;\longrightarrow\; x = 5
\]
\end{tcolorbox}

\vspace{0.1cm}
\noindent\textbf{Practica:}

\begin{exercicis}
    \item $3^x = 81$
    \item $7^x = 49$
    \item $2^{x+1} = 16$
    \item $5^{2x} = 125$
    \item $10^{x-2} = 1\,000$
    \item $10^{x+3} = 10000$
    \item $7^{3x-2} = 49$
    \item $2^{x/2} = 8$
    \item $3^{x^2-1} = 27$
\end{exercicis}

\secciotipus{Reescriure ambdĆ³s costats amb la mateixa base}
% ========================================================

\begin{metodebox}
\textbf{MĆØtode:} Les dues bases no coincideixen, perĆ² \textbf{totes dues es poden expressar com a potĆØncies d'una base comĆŗ} (ex.\ $4=2^2$, $8=2^3$, $9=3^2$, $27=3^3$\ldots). Reescrivim, apliquem $(a^m)^n = a^{mn}$ i igualem els exponents.
\end{metodebox}

\vspace{0.3cm}

\begin{tcolorbox}[exemplebox]
Resol $\;4^x = 8$
\tcblower
Expressem $4 = 2^2$ i $8 = 2^3$:
\[
(2^2)^x = 2^3 \;\longrightarrow\; 2^{2x} = 2^3 \;\longrightarrow\; 2x = 3 \;\longrightarrow\; x = \frac{3}{2}
\]
\end{tcolorbox}

\vspace{0.4cm}
\noindent\textbf{Practica:}

\begin{exercicis}
    \item $9^x = 27$
    \item $4^{x+1} = 32$
    \item $8^{x-1} = 16$
    \item $25^x = 125$
    \item $16^x = 8^{x+2}$
    \item $16^x = 32$
    \item $4^{x-1} = 2$
    \item $9^{2x} = 243$
    \item $125^x = 25$
\end{exercicis}

\secciotipus{Suma de potĆØncies de la mateixa base --- factor comĆŗ}
% ========================================================

\begin{metodebox}
\textbf{MĆØtode:} Tenim una \textbf{suma de potĆØncies de la mateixa base} $a$. Escrivim totes les potĆØncies en funciĆ³ de $a^x$ (recordant que $a^{x+k}=a^x \cdot a^k$ i $a^{x-k}=\frac{a^x}{a^k}$), traiem $a^x$ com a \textbf{factor comĆŗ}, simplifiquem el parĆØntesi i aĆÆllem $a^x$ per igualar finalment els exponents.
\end{metodebox}

\vspace{0.3cm}

\begin{tcolorbox}[exemplebox]
Resol $\;2^{x-1} + 2^x + 2^{x+1} = 28$
\tcblower
Reescrivim cada terme en funciĆ³ de $2^x$:
\[
\frac{2^x}{2} + 2^x + 2^x \cdot 2 = 28
\]
Treiem $2^x$ com a factor comĆŗ:
\[
2^x \!\left(\frac{1}{2} + 1 + 2\right) = 28 \;\longrightarrow\; 2^x \cdot \frac{7}{2} = 28 \;\longrightarrow\; 2^x = 8 = 2^3 \;\longrightarrow\; x = 3
\]
\end{tcolorbox}

\vspace{0.4cm}
\noindent\textbf{Practica:}

\begin{exercicis}
    \item $3^{x-1} + 3^x + 3^{x+1} = 13$
    \item $5^x + 5^{x+1} = 30$
    \item $2^x + 2^{x+1} + 2^{x+2} = 56$
    \item $3^{x-1} + 3^x = 12$
    \item $2^x + 2^{x+1} + 2^{x+2} + 2^{x+3} = 120$
    \item $2^x + 2^{x+2} = 40$

    \item $4^x - 5 \cdot 2^x + 4 = 0$
    \item $9^x - 12 \cdot 3^x + 27 = 0$
    \item $25^x - 30 \cdot 5^x + 125 = 0$
\end{exercicis}

\secciotipus{Canvi de variable --- equaciĆ³ de segon grau en $a^x$}
% ========================================================

\begin{metodebox}
\textbf{MĆØtode:} L'equaciĆ³ contĆ© $a^{2x}$ i $a^x$. Com que $a^{2x} = (a^x)^2$, fem el \textbf{canvi de variable} $t = a^x$ per obtenir una equaciĆ³ de segon grau en $t$. Resolem per $t$ i desfem el canvi ($a^x = t_i$, expressem $t_i$ com a potĆØncia de $a$ i igualem exponents). \textbf{Descartem} les solucions negatives o zero (ja que $a^x > 0$).
\end{metodebox}

\vspace{0.3cm}

\begin{tcolorbox}[exemplebox]
Resol $\;4^x - 5 \cdot 2^x + 4 = 0$
\tcblower
Com que $4^x = (2^2)^x = (2^x)^2$, fem el canvi $t = 2^x$ ($t > 0$):
\[
t^2 - 5t + 4 = 0 \;\longrightarrow\; (t-1)(t-4) = 0
\]
\[
t_1 = 1 \;\Rightarrow\; 2^x = 1 = 2^0 \;\Rightarrow\; x_1 = 0
\]
\[
t_2 = 4 \;\Rightarrow\; 2^x = 4 = 2^2 \;\Rightarrow\; x_2 = 2
\]
\end{tcolorbox}

\vspace{0.4cm}
\noindent\textbf{Practica:}

\begin{exercicis}
    \item $4^x - 3 \cdot 2^x - 4 = 0$
    \item $9^x - 4 \cdot 3^x + 3 = 0$

\secciotipus{SoluciĆ³ numĆØrica per aproximaciĆ³}

\begin{metodebox}

    \item Anem afÄ±nant el valor de l'exponent decimal fins al nombre de decimals demanat.
    \item Donem la \textbf{soluciĆ³ per defecte} (truncament) i la \textbf{soluciĆ³ per excĆ©s}.
\end{enumerate}
\end{metodebox}

\vspace{0.3cm}

\begin{tcolorbox}[exemplebox]

\tcblower

\vspace{0.3cm}
\textbf{Pas 2.} Afinem provant valors intermedis:

\medskip
\begin{center}

\end{tabular}
\end{center}

\vspace{0.2cm}
\textbf{Pas 3.} Conclusions:
\[
\text{SoluciĆ³ per defecte (2 dec.):}\; x \approx 3{,}32 \qquad
\text{SoluciĆ³ per excĆ©s (2 dec.):}\; x \approx 3{,}33
\]
\end{tcolorbox}
